### 1.2 技术栈选型

| 层级 | 技术 | 选型理由 | 替代方案 |
|------|------|----------|----------|
| **前端框架** | Next.js 14 (App Router) | SSR/SEO 友好，Vercel 一键部署 | Nuxt.js / Remix |
| **UI 组件库** | Shadcn/ui + TailwindCSS | 高度可定制，组件质量高 | Ant Design / MUI |
| **图表库** | Recharts / Tremor | 轻量，适合 Dashboard | ECharts / Chart.js |
| **后端框架** | Hono (Edge) 或 FastAPI | Hono 边缘部署低延迟；FastAPI 生态成熟 | Express / NestJS |
| **数据库** | PostgreSQL (Supabase) | 开源，支持 Row Level Security | PlanetScale / Neon |
| **缓存/队列** | Upstash Redis | Serverless 兼容，按量付费 | Redis Cloud |
| **认证** | Clerk / Supabase Auth | 开箱即用，支持多订阅等级 | Auth0 / NextAuth |
| **支付** | Stripe / 支付宝 (当面付) | 订阅管理完善 | Lemon Squeezy / Paddle |
| **邮件** | Resend / SendGrid | 送达率高，开发者友好 | Postmark |
| **部署** | Vercel (前端) + Railway (后端) | 零运维，自动 CI/CD | Cloudflare Pages + Workers |
| **监控** | Sentry + Vercel Analytics | 错误追踪 + 性能监控 | LogRocket |

### 1.3 数据库设计

```sql
-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    avatar_url VARCHAR(500),
    subscription_tier VARCHAR(50) DEFAULT 'free', -- free/pro/business/enterprise
    subscription_status VARCHAR(50) DEFAULT 'active',
    stripe_customer_id VARCHAR(255),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 热点事件表
CREATE TABLE hotspot_events (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    event_title VARCHAR(500) NOT NULL,
    event_summary TEXT,
    event_type VARCHAR(100), -- 政策/灾害/科技/消费...
    source_platform VARCHAR(100), -- 微博/抖音/百度...
    heat_score DECIMAL(5,2), -- 热度分数
    stage VARCHAR(50), -- 爆发期/扩散期/发酵期/回落期
    raw_data JSONB, -- 原始 API 数据
    fetched_at TIMESTAMP DEFAULT NOW(),
    INDEX idx_heat_score (heat_score DESC),
    INDEX idx_stage (stage)
);

-- 需求预测结果表
CREATE TABLE need_predictions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    event_id UUID REFERENCES hotspot_events(id),
    segment_id VARCHAR(100), -- 人群细分
    need_id VARCHAR(100),
    need_name VARCHAR(255),
    sub_need_name VARCHAR(255),
    score DECIMAL(5,4),
    urgency DECIMAL(5,4),
    confidence DECIMAL(5,4),
    reasoning JSONB, -- 推理链
    is_premium BOOLEAN DEFAULT FALSE, -- 是否付费可见
    created_at TIMESTAMP DEFAULT NOW(),
    INDEX idx_event_id (event_id),
    INDEX idx_score (score DESC)
);

-- 用户查询历史表
CREATE TABLE user_queries (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    query_text TEXT NOT NULL,
    prediction_result JSONB,
    credits_used INTEGER DEFAULT 1,
    created_at TIMESTAMP DEFAULT NOW(),
    INDEX idx_user_id (user_id),
    INDEX idx_created_at (created_at DESC)
);

-- 订阅计划表
CREATE TABLE subscription_plans (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL, -- free/pro/business
    price_monthly INTEGER, -- 分
    price_yearly INTEGER,
    daily_queries_limit INTEGER,
    api_access BOOLEAN DEFAULT FALSE,
    custom_segments BOOLEAN DEFAULT FALSE,
    email_reports BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 用户每日配额表
CREATE TABLE user_daily_quota (
    user_id UUID REFERENCES users(id),
    date DATE DEFAULT CURRENT_DATE,
    queries_used INTEGER DEFAULT 0,
    queries_limit INTEGER DEFAULT 3,
    reset_at TIMESTAMP DEFAULT NOW() + INTERVAL '1 day',
    PRIMARY KEY (user_id, date)
);

-- 邮件简报订阅表
CREATE TABLE newsletter_subscriptions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    subscribed_at TIMESTAMP DEFAULT NOW(),
    unsubscribed_at TIMESTAMP,
    preferences JSONB -- 偏好的事件类型/人群
);
```

### 1.5 限流与配额设计

```python
# 配额中间件伪代码

SUBSCRIPTION_LIMITS = {
    "free": {
        "daily_queries": 3,
        "api_calls_per_minute": 5,
        "history_days": 7,
        "email_reports": False,
        "api_access": False,
    },
    "pro": {
        "daily_queries": 50,
        "api_calls_per_minute": 30,
        "history_days": 90,
        "email_reports": True,
        "api_access": False,
    },
    "business": {
        "daily_queries": 500,
        "api_calls_per_minute": 100,
        "history_days": 365,
        "email_reports": True,
        "api_access": True,
    },
    "enterprise": {
        "daily_queries": -1,  # 无限
        "api_calls_per_minute": 500,
        "history_days": -1,
        "email_reports": True,
        "api_access": True,
    },
}

async def quota_middleware(request, call_next):
    user = await get_current_user(request)
    if not user:
        return JSONResponse({"error": "Unauthorized"}, status_code=401)
    
    limits = SUBSCRIPTION_LIMITS[user.subscription_tier]
    today = datetime.now().date()
    
    # 检查今日配额
    quota = await db.user_daily_quota.get(user.id, today)
    if quota.queries_used >= limits["daily_queries"]:
        return JSONResponse(
            {"error": "Daily quota exceeded", "upgrade_url": "/pricing"},
            status_code=429
        )
    
    # 执行请求
    response = await call_next(request)
    
    # 更新配额
    if request.method == "POST" and "/predict" in request.url.path:
        await db.user_daily_quota.increment(user.id, today)
    
    # 添加配额头
    response.headers["X-Quota-Used"] = str(quota.queries_used)
    response.headers["X-Quota-Limit"] = str(limits["daily_queries"])
    
    return response
```
